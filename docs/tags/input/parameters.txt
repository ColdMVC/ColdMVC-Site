| class        | false    | string  |                   | The class for the input.                   |

| instructions | false    | string  |                   | The instructions for the input.            |

| onblur       | false    | string  |                   | The onblur event for the input.            |
| onchange     | false    | string  |                   | The onchange event for the input.          |
| onclick      | false    | string  |                   | The onclick event for the input.           |
| ondblclick   | false    | string  |                   | The ondblclick event for the input.        |
| onfocus      | false    | string  |                   | The onfocus event for the input.           |
| onkeydown    | false    | string  |                   | The onkeydown event for the input.         |

| onkeyup      | false    | string  |                   | The onkeyup event for the input.           |
| onsubmit     | false    | string  |                   | The onsubmit event for the input.          |
| readonly     | false    | boolean | false             | If the input is read only.                 |
| style        | false    | string  |                   | The style for the input.                   |
| title        | false    | string  | ${label}          | The title for the input.                   |
| value        | false    | string  |                   | The value of the input.                    |
